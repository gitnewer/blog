git 初始化
git init

添加文件
git add  filename

提交git
git commit -m 'xxxxxx'

查看状态
git status
git diff
git log

退回修改
git checkout -- filename 

文件删除
git rm 

远程
git push origin master
git clone git@github.com：name/xxxx.git

创建分支
git checkout -b dev
git branch

合并分支
git merge dev
